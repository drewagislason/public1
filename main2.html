Example: FlyJsonPut
char szName[32] = "My Name";
int age = 38;
char szOutStr[100];
const flyJsonPath_t aJsonPaths[] = {
{ "name", szName, FLYJSON_STR },
{ "age", &age, FLYJSON_INT },
};

FlyJsonPut(szOutStr, sizeof(szOutStr), aJsonPaths, NumElements(aJsonPaths));
printf("%s\n", szOutStr); // prints "{ "name": "My Name", "age": 38 }"
